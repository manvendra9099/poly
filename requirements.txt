# btcfm runtime requirements
#
# On CCuB, do NOT run: pip install -r requirements.txt
# Use setup_ccub.sh instead, which installs everything in ONE pip invocation.
#
# WHY ONE INVOCATION?
#   If jax[cuda12_pip]==0.4.30 is installed first and -r requirements.txt
#   runs separately, pip re-resolves from scratch.  It then sees:
#     optax 0.2.8  → jaxlib>=0.5.3  (conflicts with jaxlib 0.4.30)
#   and upgrades jaxlib to the CPU 0.7.0 build, silently discarding CUDA.
#   A single pip call sees jax==0.4.30's jaxlib<=0.4.30 constraint from the
#   start and picks versions of optax/flax that are compatible with it.
#
# KEY VERSION CONSTRAINTS (all enforced in setup_ccub.sh):
#   jax[cuda12_pip]==0.4.30    — last JAX 0.4.x with cuda12_pip extra
#   flax>=0.8.0,<0.9.0         — 0.9.0+ requires newer jaxlib
#   orbax-checkpoint<0.5.0     — 0.5.0+ requires tensorstore (Bazel/GLIBC 2.25)
#   pyarrow>=14.0,<18.0        — 18.0 dropped cp311 manylinux2014 wheels

flax>=0.8.0,<0.9.0
orbax-checkpoint<0.5.0
optax>=0.2.0
numpy>=1.26
polars>=1.40,<2
pyarrow>=14.0,<18.0
requests>=2.31
